# Logos

<ArgsTable story="Brown" />

### Brown and CCV logos components:
```jsx
<BrownLogo />
```
```jsx
<CCVLogo />
```

Logo components take the prop `size` and accepts the following values:
`'xxs', 'xs', 's', 'm', 'l', 'xl', 'xxl'`.


export const Template1 = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { BrownLogo },
  template: `<BrownLogo :size="size"/>`
});

export const Template2 = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { CCVLogo },
  template: `<CCVLogo :size="size"/>`
});
